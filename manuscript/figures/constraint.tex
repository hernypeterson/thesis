% view angle ({zenith}{azimuth})
\pgfmathsetmacro{\viewze}{65}
\pgfmathsetmacro{\viewaz}{105}
\tdplotsetmaincoords{\viewze}{\viewaz}

% define lengths
\pgfmathsetmacro{\radius}{1.3}
\pgfmathsetmacro{\axislength}{1.4}

% define angles for p
\pgfmathsetmacro{\az}{40}
\pgfmathsetmacro{\ze}{37}

% define angles for p' (no ' in macros!)
\pgfmathsetmacro{\azi}{330}
\pgfmathsetmacro{\zen}{14.4523587}

\pgfmathsetmacro{\cenang}{34.438487775}

\begin{tikzpicture}[scale=5,tdplot_main_coords]

% origin
\coordinate (O) at (0,0,0);

% also defines (Pxy), (Pxz), (Pyz), etc.
\tdplotsetcoord{P}{\radius}{\ze}{\az}
\tdplotsetcoord{P'}{\radius}{\zen}{\azi}
\tdplotsetcoord{Q}{\radius}{90}{\azi}

% grey hemisphere
\filldraw[tdplot_screen_coords, color = gray, opacity = 0.1] (0,0) -- (\radius,0) arc (0:180:\radius) -- (0,0);
\filldraw[tdplot_main_coords, color = gray!10!white] (0,0) -- (0:\radius) arc (0:365:\radius) -- (0,0);

% circumference
\tdplotdrawarc[very thin, dashed]{(O)}{\radius}{\viewaz}{\viewaz+180}{}{}
\tdplotdrawarc[very thin]{(O)}{\radius}{\viewaz-180}{\viewaz}{}{}

% az' surface
\tdplotsetthetaplanecoords{\azi}

% fill az' surface
\filldraw[tdplot_rotated_coords, color = red, opacity = 0.45] (\radius,0) arc (0:\zen:\radius) -- (0,0);
\filldraw[tdplot_rotated_coords, color = black, opacity = 0.05] (\radius,0) arc (0:90:\radius) -- (0,0);

% line az' surface
\draw[very thin] (O) -- (P') node[anchor = south east] {\acs{closest-point}};
\draw[very thin] (O) -- (Q);
\tdplotdrawarc[tdplot_rotated_coords, very thick]{(O)}{\radius}{90}{14}{anchor=north west}{\acs{line}}
\tdplotdrawarc[tdplot_rotated_coords,very thick,red]{(O)}{\radius}{0}{\zen}{anchor=south}{\acs{ze'}}

% az surface
\tdplotsetthetaplanecoords{\az}

% line az surface
\draw[very thin] (P) -- (O) -- (Pxy) -- (P) node[anchor = south west] {\acs{point}};

% cartesian axes
\draw[arrow] (O) -- (\axislength,0,0) node[anchor=north]{$x$};
\draw[arrow] (O) -- (0,\axislength,0) node[anchor=west]{$y$};
\draw[arrow] (O) -- (0,0,\axislength) node[anchor=south]{$z$};

% az' angle label
\tdplotdrawarc{(O)}{0.15}{0}{\azi}{anchor=south west}{\acs{az'}}

% az angle label
\tdplotdrawarc{(O)}{0.25}{0}{\az}{anchor=north}{\acs{az}}

% ze angle label
\tdplotdrawarc[tdplot_rotated_coords]{(O)}{0.3}{0}{\ze}{anchor=south west}{\acs{ze}}

% central angle plane
\tdplotsetrotatedcoords{\azi}{-90+\zen}{0}

% central angle line
\tdplotdrawarc[tdplot_rotated_coords,very thick,blue]{(0,0,0)}{\radius}{\cenang}{0}{anchor=south west}{\acs{central-angle}}

% central angle fill
\filldraw[tdplot_rotated_coords, color = blue, opacity = 0.15] (\radius,0) arc (0:\cenang:\radius) -- (0,0);

% draw points p and p' 
\fill (P) circle [radius=0.4pt];
\fill[red] (P') circle [radius=0.4pt];

\end{tikzpicture}
