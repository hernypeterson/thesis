\begin{tikzpicture}[scale=5.3,tdplot_main_coords]

% origin
\coordinate (O) at (0,0,0);

% also defines (Pxy), (Pxz), (Pyz), etc.
\tdplotsetcoord{P}{\radius}{\ze}{\az}

% fill flat surface
\filldraw[color = gray!10!white] (0,0) -- (0:\radius) arc (0:360:\radius) -- (0,0);

% define tilted surface
\tdplotsetrotatedcoords{\az}{\ze}{0}

% fill tilted surface
\filldraw[tdplot_rotated_coords, color = green!40!black, opacity=0.2] (0,0) -- (0:\radius) arc (0:360:\radius) -- (0,0);

% tilted axes
\draw[tdplot_rotated_coords, color = green!40!black, arrow] (0:-\radius) -- (0:\radius);
\draw[tdplot_rotated_coords, color = green!40!black, arrow] (90:-\radius) -- (270:\radius);

% horizontal surface (front)
\filldraw[color = gray!10!white, opacity=0.7] (\az-90:\radius) arc (\az-90:\az+90:\radius);
% horizontal circumference (front)
\tdplotdrawarc[very thin]{(O)}{\radius}{\az+90}{\az-\viewaz-25}{}{}

% tilted circumference (front)
\tdplotdrawarc[tdplot_rotated_coords, very thin, color = green!40!black]{(O)}{\radius}{\az+31}{\az-\viewaz-25}{}{}

% az surface
\tdplotsetthetaplanecoords{\az}

% vector p
\draw[very thin, dashed,green!40!black] (P) -- (Pxy) -- (O) -- (P);
\draw[arrow,green!40!black] (O) -- (P) node[anchor = south west] {\acs{point}};

% cartesian axes
\draw[arrow] (O) -- (\axislength,0,0) node[anchor=north]{$x$};
\draw[arrow] (O) -- (0,\axislength,0) node[anchor=west]{$y$};
\draw[arrow] (O) -- (0,0,\axislength) node[anchor=south]{$z$};

% az angle label
\tdplotdrawarc{(O)}{0.25}{0}{\az}{anchor=north}{\acs{az}}

% ze angle label
\tdplotdrawarc[tdplot_rotated_coords]{(O)}{0.3}{0}{\ze}{anchor=south west}{\acs{ze}}

% cartesian axes
\draw[green!40!black, arrow,tdplot_rotated_coords] (0,0,0) -- (\axislength,0,0);
\draw[green!40!black, arrow,tdplot_rotated_coords] (0,-\axislength,0) -- (0,\axislength,0);
% \draw[arrow,tdplot_rotated_coords] (0,0,-\axislength) -- (0,0,\axislength) node[anchor=south]{$z$};
\end{tikzpicture}
